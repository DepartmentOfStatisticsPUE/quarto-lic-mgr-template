% Font: Carlito (metric-compatible with Calibri)
% Requires carlito package in TeX distribution (TinyTeX: tlmgr install carlito)
\setmainfont{Carlito-Regular.ttf}[
  BoldFont = Carlito-Bold.ttf,
  ItalicFont = Carlito-Italic.ttf,
  BoldItalicFont = Carlito-BoldItalic.ttf
]

% Paragraph indentation (including first paragraph): 0.5 cm
\usepackage{indentfirst}
\setlength{\parindent}{0.5cm}

% Page style: page numbers at the bottom center
\pagestyle{plain}

% In Quarto book mode: # = \chapter, ## = \section, ### = \subsection

% Chapter headings: 14pt bold (§3.3.8 zarządzenia)
\usepackage{titlesec}
\titleformat{\chapter}[display]
  {\normalfont\fontsize{14}{16.8}\bfseries}{\chaptertitlename\ \thechapter}{0pt}{}
\titleformat{\section}
  {\normalfont\fontsize{12}{14.4}\bfseries}{\thesection}{1em}{}
\titleformat{\subsection}
  {\normalfont\fontsize{12}{14.4}\bfseries}{\thesubsection}{1em}{}

% Figure and table numbering within chapters (1.1, 1.2, 2.1, ...)
\usepackage{amsmath}
\numberwithin{figure}{chapter}
\numberwithin{table}{chapter}
\numberwithin{equation}{chapter}

% Caption styling: bold 12pt label (§3.3.10-11)
\usepackage[font=normal,labelfont=bf]{caption}

% Footnotes: 10pt (§3.3.12)
\usepackage{footnote}
\renewcommand{\footnotesize}{\fontsize{10}{12}\selectfont}

% Graphics support
\usepackage{graphicx}
\usepackage{eso-pic}
\usepackage{changepage}

% Language support (babel loaded by Quarto via lang setting)
\usepackage{csquotes}

% Biblatex customizations (deferred until biblatex is loaded)
\AtBeginDocument{%
  \renewcommand\cite{\parencite}%
  \renewcommand\&{\bibstring{and}}%
  \DeclareCiteCommand{\citepage}[\mkbibparens]
    {\usebibmacro{cite:init}}
    {\usebibmacro{cite:post}}
    {}
    {\usebibmacro{postnote}}%
}
